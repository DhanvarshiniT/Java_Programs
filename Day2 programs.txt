12.A program to calculate the electricity bill based on the number of units consumed.

import java.util.*;

public class Main {
    public static void main(String[] args) {
        Scanner sc = new Scanner(System.in);
        double units = sc.nextDouble();
        double bill = 0;

        if (units <= 50) {
            bill = units * 1;
        } else if (units <= 150) {
            bill = 50 * 1 + (units - 50) * 2;
        } else if (units <= 250) {
            bill = 50 * 1 + 100 * 2 + (units - 150) * 3;
        } else {
            bill = 50 * 1 + 100 * 2 + 100 * 3 + (units - 250) * 4;
        }

        if (bill > 150) {
            bill += bill * 0.2; 
        }

        System.out.println((int) bill);
    }
}

14.A program to calculate the train fare based on the passenger’s age and distance to travel.

import java.util.Scanner;

public class TrainFareCalculator {
    public static void main(String[] args) {
        Scanner sc = new Scanner(System.in);
        int age = sc.nextInt();
        int distance = sc.nextInt();

        int standardFarePerKm = 5;
        double totalFare = distance * standardFarePerKm;
        if (age < 12) {
            totalFare *= 0.5; 
        } else if (age >= 60) {
            totalFare *= 0.75; 
        } 
        if (totalFare == (int) totalFare) {
            System.out.println((int) totalFare);
        } else {
            System.out.println(totalFare);
        }

        sc.close();
    }
}

16.Fun Character Case Converter

public class CaseConverter {
    public static void main(String[] args) {
        Scanner sc = new Scanner(System.in);
        char ch = sc.next().charAt(0); 
        if (ch >= 'a' && ch <= 'z') {  
            System.out.println((char)(ch - 32));
        } 
        else if (ch >= 'A' && ch <= 'Z') {  
            System.out.println((char)(ch + 32));
        } 
        else {  
            System.out.println("abcd theriyatha?");
        }
        
        sc.close();
    }
